<template>
  <div>
    <etu-select
      v-model="value"
      filterable
      remote
      placeholder="Please enter a keyword"
      remote-show-suffix
      :remote-method="remoteMethod"
      :loading="loading"
    >
      <etu-option
        v-for="item in options"
        :key="item.label"
        :label="item.label"
        :value="item.value"
      ></etu-option>
    </etu-select>
  </div>
</template>

<script setup lang="ts">
import { onMounted, ref, watch } from "vue";
interface ListItem {
  value: string;
  label: string;
}

const list = ref<ListItem[]>([]);
const options = ref<ListItem[]>([]);
const value = ref("Beijing");
